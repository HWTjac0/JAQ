41double GeoLocator::haversineDistance(double lat1, double lon1, double lat2, double lon2) {
42 const double r = 6378;
43 lat1 = toRadians(lat1);
44 lon1 = toRadians(lon1);
45 lat2 = toRadians(lat2);
46 lon2 = toRadians(lon2);
47
48 const double deltaLat = lat2 - lat1;
49 const double deltaLon = lon2 - lon1;
50
51 using namespace std;
52 const double a = pow(sin(deltaLat/2), 2) + cos(lat1) * cos(lat2) * pow(sin(deltaLon/2), 2);
53 const double c = 2 * atan2(sqrt(a), sqrt(1 - a));
54
55 return r * c;
56}

71void GeoLocator::positionUpdated(const QGeoPositionInfo &info) {
72 _requestingUserLocation = false;
73 qDebug() << "GeoLocator::positionUpdated";
74 if (info.isValid()) {
75 double userLat =info.coordinate().latitude();
76 double userLon = info.coordinate().longitude();
77 double minDistance = std::numeric_limits<double>::max();
78 int nearestStationId;
79 int nearestCityId;
80 for (const auto& city : Database::index.values()) {
81 for (const auto& station : city->getStations()) {
82 double distance = haversineDistance(userLat, userLon, station.getCoords().x, station.getCoords().y);
83 if (distance < minDistance) {
84 minDistance = distance;
85 nearestStationId = station.id();
86 nearestCityId = city->id();
87 }
88 }
89 }
90 AppContext::getInstance()->getCityHandler()->changeCity(nearestCityId);
91 AppContext::getInstance()->getStationHandler()->stationSelected(nearestStationId);
92 emit userLocationAcquired(userLat, userLon);
93 }
94}
