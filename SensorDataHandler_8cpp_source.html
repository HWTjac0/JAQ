40void SensorDataHandler::onDataLoaded(const QJsonArray &data) {
41 _measurements.clear();
42 double min = 9999999;
43 double max = 0;
44 for (auto d : data) {
45 QJsonObject mobj = d.toObject();
46 QDateTime timestamp = QDateTime::fromString(
47 mobj.value("Data").toString(),
48 "yyyy-MM-dd HH:mm:ss"
49 );
50 double value = mobj.value("Wartość").toDouble();
51 min = min < value ? min : value;
52 max = max > value ? max : value;
53
54 _measurements.emplace_back(timestamp, value);
55 }
56 _sensorDataModel->setData(_measurements);
57 emit dataChanged();
58}
