29QVariant SensorDataModel::data(const QModelIndex &index, int role) const {
30 if (!index.isValid() || index.row() >= _data.size())
31 return {};
32
33 const auto &entry = _data.at(index.row());
34 const QDateTime &timestamp = entry.first;
35 const double value = entry.second;
36
37 switch (role) {
38 case TimeStampRole:
39 return timestamp;
40 case ValueRole:
41 return value;
42 default:
43 return {};
44 }
45}

66void SensorDataModel::setData(const QVector<QPair<QDateTime, double>> &data) {
67 beginResetModel();
68 _data = data;
69
70 _timerangeStart = _data.back().first;
71 _timerangeEnd = _data.front().first;
72 auto mm = std::minmax_element(_data.begin(), _data.end(), [](const QPair<QDateTime, double> &a, const QPair<QDateTime, double> &b) {
73 return a.second < b.second;
74 });
75 _minValue = (*mm.first).second;
76 _maxValue = (*mm.second).second;
77
78 endResetModel();
79}
