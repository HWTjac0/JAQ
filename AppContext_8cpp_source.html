22void AppContext::initialize() {
23 _networkChecker.reset(new NetworkChecker());
24 connect(_networkChecker.get(), &NetworkChecker::connectionStatusChanged, this, &AppContext::networkStatusChanged, Qt::QueuedConnection);
25 _networkChecker->startChecking(5000);
26 _apiClient.reset(new ApiClient(this));
27 _database.reset(new Database(_apiClient.get(), this));
28 connect(_database.get(), &Database::databaseReady, this, [=]() {
29 qDebug() << "AppContext::initialize";
30 setupHandlers();
31 emit initialized();
32 });
33 _database.get()->init();
34}

36void AppContext::setupHandlers() {
37 /**
38 * I am passing api client, handlers etc. explicitly because I think
39 * implicit access to them is kind of convoluted for this init usage

41 _sensorHandler.reset(new SensorHandler(_apiClient.get(), this));
42 _sensorDataHandler.reset(new SensorDataHandler(_apiClient.get(), this));
43 _stationHandler.reset(new StationHandler(_apiClient.get(),_sensorHandler.get(),this));
44 _cityHandler.reset(new CityHandler(_apiClient.get(), _stationHandler.get(), this));
45

48 &CityHandler::cityChanged,
49 _sensorHandler.get(),
